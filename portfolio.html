Chemokine Receptor Expression Analysis

Python · Data Visualization · Spatial Transcriptomics


import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd

chemokine_receptors = ["Cd69","Itga1","Itgae","Cxcr6","Ly6c1","Ly6c2","Cd226"]
chemokine_receptors = [gene.upper() for gene in chemokine_receptors]

genes_in_dataset = [gene for gene in chemokine_receptors if gene in all_data_filtered_celltype.var_names]
missing_genes = [gene for gene in chemokine_receptors if gene not in all_data_filtered_celltype.var_names]

print("Genes present in the dataset:", genes_in_dataset)
print("Missing genes:", missing_genes)

df = all_data_filtered_celltype.to_df()
df['condition'] = all_data_filtered_celltype.obs['condition']
df['celltype_minor'] = all_data_filtered_celltype.obs['celltype_minor']

df_receptors = df[genes_in_dataset + ['condition', 'celltype_minor']]
df_receptors_melted = df_receptors.melt(id_vars=['condition', 'celltype_minor'],
                                        value_vars=genes_in_dataset,
                                        var_name='Chemokine_Receptor',
                                        value_name='Expression')

plt.figure(figsize=(50, 80))
sns.set(style="whitegrid")
unique_receptors = df_receptors_melted['Chemokine_Receptor'].unique()

for i, receptor in enumerate(unique_receptors[:20]):
    receptor_data = df_receptors_melted[df_receptors_melted['Chemokine_Receptor'] == receptor]
    plt.subplot(10, 2, i + 1)
    sns.violinplot(x="celltype_minor", y="Expression", hue="condition", data=receptor_data,
                   palette="Set2", inner="quart", density_norm='width', linewidth=1.2, split=False)
    plt.title(f"{receptor} Expression Across Cell Types", fontsize=18)
    plt.xlabel('Cell Type', fontsize=14)
    plt.ylabel('Expression', fontsize=14)
    plt.xticks(rotation=45, ha='right', fontsize=12)
    plt.yticks(fontsize=12)
    plt.grid(True, linestyle='--', alpha=0.6)

plt.subplots_adjust(hspace=0.4)
plt.show()
